import math
print('''Choose your geometrical figure:
        1. Circle
        2. Square
        3. Rectangle''')
n=input("Choose your option: ")
while n not in "123":
    print("Please enter a valid option")
    n=input("Choose your option: ")
if n==1:
    a=float(input("Enter the radius: "))
    res=math.pi()*(a**2)
elif n==2:
    a=float(input("Enter the length of each side: "))
    res=a**2
elif n==3:
    l=float(input("Enter the length: "))
    b=float(input("Enter the breadth: "))
    res=l*b
print("The area is",res)
